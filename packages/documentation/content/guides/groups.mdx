# Working with Groups

Learn how to organize multiple MoneyWorks companies into groups for efficient management by accounting firms and multi-entity organizations.

## Overview

Groups in MoneyWorks allow you to:

- **🏢 Multi-Company Management** - Handle multiple client files from one interface
- **👥 User Access Control** - Assign users to specific groups and companies
- **📊 Consolidated Reporting** - Generate reports across multiple entities
- **🔐 Segregated Access** - Keep client data isolated and secure
- **🚀 Efficient Workflows** - Switch between companies seamlessly
- **📁 Organized Structure** - Hierarchical organization of related entities

## Working with Multiple Companies

### Company Switching

Implement seamless switching between companies:

<CodeExample>
```typescript
import { MoneyWorksClient } from '@moneyworks/data';

class MultiCompanyManager {
  private connections = new Map<string, MoneyWorksClient>();
  private currentCompany?: string;
  
  async switchToCompany(companyId: string) {
    // Check if user has access
    const hasAccess = await this.checkAccess(companyId);
    if (!hasAccess) {
      throw new Error('Access denied to company');
    }
    
    // Get or create connection
    let client = this.connections.get(companyId);
    
    if (!client) {
      const config = await this.getCompanyConfig(companyId);
      client = new MoneyWorksClient(config);
      await client.connect();
      this.connections.set(companyId, client);
    }
    
    this.currentCompany = companyId;
    return client;
  }
  
  getCurrentClient(): MoneyWorksClient {
    if (!this.currentCompany) {
      throw new Error('No company selected');
    }
    
    const client = this.connections.get(this.currentCompany);
    if (!client) {
      throw new Error('Company not connected');
    }
    
    return client;
  }
  
  async closeAll() {
    for (const [companyId, client] of this.connections) {
      await client.disconnect();
    }
    this.connections.clear();
  }
}
```
</CodeExample>

### Batch Operations

Perform operations across multiple companies:

<CodeExample>

```typescript
// Export tax rates from all companies in a group
async function exportGroupTaxRates(groupId: string) {
  const companies = await getCompaniesInGroup(groupId);
  const results = new Map<string, any>();
  
  for (const company of companies) {
    try {
      const client = await switchToCompany(company.id);
      const taxRates = await client.export('TaxRate', {
        format: 'full'
      });
      
      results.set(company.id, {
        companyName: company.name,
        taxRates,
        exportedAt: new Date()
      });
    } catch (error) {
      results.set(company.id, {
        companyName: company.name,
        error: error.message
      });
    }
  }
  
  return results;
}

// Standardize chart of accounts across group
async function standardizeChartOfAccounts(
  groupId: string,
  templateCompanyId: string
) {
  // Get template accounts
  const templateClient = await switchToCompany(templateCompanyId);
  const templateAccounts = await templateClient.export('Account', {
    format: 'full'
  });
  
  const companies = await getCompaniesInGroup(groupId);
  const results = [];
  
  for (const company of companies) {
    if (company.id === templateCompanyId) continue;
    
    try {
      const client = await switchToCompany(company.id);
      
      // Import accounts (implementation depends on MW API)
      await client.import('Account', templateAccounts, {
        updateExisting: false,
        skipDuplicates: true
      });
      
      results.push({
        companyId: company.id,
        status: 'success',
        accountsImported: templateAccounts.length
      });
    } catch (error) {
      results.push({
        companyId: company.id,
        status: 'error',
        error: error.message
      });
    }
  }
  
  return results;
}
```

## Consolidated Reporting

### Cross-Company Reports

Generate reports across multiple entities:

<CodeExample>
```typescript
interface ConsolidatedReport {
  groupId: string;
  reportType: string;
  period: Period;
  companies: CompanyReport[];
  totals: ReportTotals;
  generatedAt: Date;
}

interface CompanyReport {
  companyId: string;
  companyName: string;
  data: any;
  errors?: string[];
}

async function generateConsolidatedPL(
  groupId: string,
  period: Period
) {
  const companies = await getCompaniesInGroup(groupId);
  const report: ConsolidatedReport = {
    groupId,
    reportType: 'profit_loss',
    period,
    companies: [],
    totals: {
      revenue: 0,
      expenses: 0,
      netIncome: 0
    },
    generatedAt: new Date()
  };
  
  // Fetch P&L for each company
  for (const company of companies) {
    try {
      const client = await switchToCompany(company.id);
      const plData = await client.runReport('ProfitLoss', {
        period: period.toString(),
        format: 'json'
      });
      
      report.companies.push({
        companyId: company.id,
        companyName: company.name,
        data: plData
      });
      
      // Aggregate totals
      report.totals.revenue += plData.revenue;
      report.totals.expenses += plData.expenses;
      report.totals.netIncome += plData.netIncome;
      
    } catch (error) {
      report.companies.push({
        companyId: company.id,
        companyName: company.name,
        data: null,
        errors: [error.message]
      });
    }
  }
  
  return report;
}
```

### Inter-Company Eliminations

Handle inter-company transactions for consolidated reporting:

<CodeExample>
```typescript
interface InterCompanyTransaction {
  fromCompany: string;
  toCompany: string;
  date: YYYYMMDD;
  amount: number;
  description: string;
  fromAccount: AccountCode;
  toAccount: AccountCode;
}

async function identifyInterCompanyTransactions(
  groupId: string,
  period: Period
): Promise<InterCompanyTransaction[]> {
  const companies = await getCompaniesInGroup(groupId);
  const icTransactions: InterCompanyTransaction[] = [];
  
  // Define inter-company account codes
  const icAccountPattern = /^2[89]\d{2}/; // 28xx, 29xx accounts
  
  for (const company of companies) {
    const client = await switchToCompany(company.id);
    
    // Find transactions to IC accounts
    const transactions = await client.export('Transaction', {
      filter: `Date >= '${period.startDate()}' AND Date <= '${period.endDate()}'`,
      format: 'full'
    });
    
    for (const txn of transactions) {
      if (icAccountPattern.test(txn.AccountCode)) {
        // Extract counterparty from account description
        const counterparty = extractCounterparty(txn.Description);
        
        if (counterparty && companies.some(c => c.id === counterparty)) {
          icTransactions.push({
            fromCompany: company.id,
            toCompany: counterparty,
            date: YYYYMMDD(txn.Date),
            amount: txn.Amount,
            description: txn.Description,
            fromAccount: AccountCode(txn.AccountCode),
            toAccount: getCounterpartyAccount(txn.AccountCode)
          });
        }
      }
    }
  }
  
  return icTransactions;
}

// Eliminate inter-company transactions from consolidated report
function eliminateInterCompany(
  consolidatedData: any,
  icTransactions: InterCompanyTransaction[]
) {
  // Remove inter-company revenues and expenses
  for (const txn of icTransactions) {
    // Reduce revenue in selling company
    consolidatedData[txn.fromCompany].revenue -= txn.amount;
    
    // Reduce expense in buying company
    consolidatedData[txn.toCompany].expenses -= txn.amount;
  }
  
  return consolidatedData;
}
```
</CodeExample>

### 3. Performance Optimization

<CodeExample>
```typescript
// Connection pooling for multi-company operations
class ConnectionPool {
  private maxConnections = 10;
  private connections = new Map<string, MoneyWorksClient>();
  private lastUsed = new Map<string, number>();
  
  async getConnection(companyId: string): Promise<MoneyWorksClient> {
    // Reuse existing connection
    if (this.connections.has(companyId)) {
      this.lastUsed.set(companyId, Date.now());
      return this.connections.get(companyId)!;
    }
    
    // Clean up old connections if at limit
    if (this.connections.size >= this.maxConnections) {
      await this.evictOldest();
    }
    
    // Create new connection
    const config = await getCompanyConfig(companyId);
    const client = new MoneyWorksClient(config);
    await client.connect();
    
    this.connections.set(companyId, client);
    this.lastUsed.set(companyId, Date.now());
    
    return client;
  }
  
  private async evictOldest() {
    let oldestTime = Date.now();
    let oldestId = '';
    
    for (const [id, time] of this.lastUsed) {
      if (time < oldestTime) {
        oldestTime = time;
        oldestId = id;
      }
    }
    
    if (oldestId) {
      const client = this.connections.get(oldestId);
      await client?.disconnect();
      this.connections.delete(oldestId);
      this.lastUsed.delete(oldestId);
    }
  }
}
```
</CodeExample>

### 4. Error Handling

Handle multi-company errors gracefully:

<CodeExample>
```typescript
interface BatchOperationResult {
  successful: string[];
  failed: Array<{
    companyId: string;
    error: string;
  }>;
  summary: {
    total: number;
    succeeded: number;
    failed: number;
  };
}

async function batchOperation(
  groupId: string,
  operation: (client: MoneyWorksClient, companyId: string) => Promise<any>
): Promise<BatchOperationResult> {
  const companies = await getCompaniesInGroup(groupId);
  const result: BatchOperationResult = {
    successful: [],
    failed: [],
    summary: {
      total: companies.length,
      succeeded: 0,
      failed: 0
    }
  };
  
  for (const company of companies) {
    try {
      const client = await getConnection(company.id);
      await operation(client, company.id);
      
      result.successful.push(company.id);
      result.summary.succeeded++;
      
    } catch (error) {
      result.failed.push({
        companyId: company.id,
        error: error.message
      });
      result.summary.failed++;
      
      // Log error but continue with other companies
      console.error(`Operation failed for ${company.id}:`, error);
    }
  }
  
  return result;
}
```
</CodeExample>
