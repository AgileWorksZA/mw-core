# Basic Usage Examples

Learn how to use the core features of @moneyworks/utilities through practical examples.

## Date Handling with YYYYMMDD

MoneyWorks uses YYYYMMDD format for all date fields. Here's how to work with them:

### Creating and Validating Dates

```typescript
import { 
  createYYYYMMDD, 
  isYYYYMMDD, 
  tryCreateYYYYMMDD,
  parseToYYYYMMDD 
} from '@moneyworks/utilities';

// Validate user input
const userInput = "20250115";
if (isYYYYMMDD(userInput)) {
  // userInput is now typed as YYYYMMDD
  console.log("Valid date:", userInput);
}

// Create with validation (throws on invalid)
try {
  const date = createYYYYMMDD("20250115");
  console.log("Created date:", date);
} catch (e) {
  console.error("Invalid date format");
}

// Safe creation (returns null on invalid)
const maybeDate = tryCreateYYYYMMDD(userInput);
if (maybeDate) {
  console.log("Valid date:", maybeDate);
}

// Parse various formats
const date1 = parseToYYYYMMDD("20250115");       // Already YYYYMMDD
const date2 = parseToYYYYMMDD("2025-01-15");     // ISO format
const date3 = parseToYYYYMMDD("2025/01/15");     // Slash format
const date4 = parseToYYYYMMDD(new Date());       // From Date object
```

### Using the Tagged Template

The `d` tagged template provides a convenient way to create dates with methods:

```typescript
import { d } from '@moneyworks/utilities';

// Create dates easily
const today = d`${new Date()}`;
const specific = d`20250115`;
const parsed = d`2025-01-15`;

// Chainable methods
const nextMonth = today.addMonths(1).addDays(15);

// Comparisons
if (specific.gt(today)) {
  console.log("Date is in the future");
}

// Get days between
const daysDiff = specific.subtract(today);
console.log(`Days difference: ${daysDiff}`);

// Format for display
console.log(today.format());     // "2025-01-15"
console.log(today.format("/"));  // "2025/01/15"
console.log(today.format("."));  // "2025.01.15"
```

## Working with Periods

Periods represent year/month combinations as numbers (YYYYMM):

```typescript
import { p, PeriodUtils } from '@moneyworks/utilities';

// Create periods
const period1 = PeriodUtils.create(202501);
const period2 = PeriodUtils.fromYearMonth(2025, 1);
const period3 = p`202501`;
const current = p`${new Date()}`;

// Extract components
const year = PeriodUtils.year(period1);   // 2025
const month = PeriodUtils.month(period1);  // 1

// Get date range for a period
const startDate = PeriodUtils.startDate(period1); // "20250101"
const endDate = PeriodUtils.endDate(period1);     // "20250131"

// Period arithmetic
const nextQuarter = PeriodUtils.addMonths(period1, 3);
const lastYear = PeriodUtils.addMonths(period1, -12);

// Generate a range of periods
const yearPeriods = PeriodUtils.range(202501, 202512);
// [202501, 202502, ..., 202512]

// Format for display
console.log(PeriodUtils.format(period1));      // "2025/01"
console.log(PeriodUtils.format(period1, "-")); // "2025-01"
```

## JSON Parsing with Type Conversion

Automatically convert MoneyWorks types when parsing JSON:

```typescript
import { parseMoneyWorksJSON } from '@moneyworks/utilities';

// Parse with automatic conversion
const data = parseMoneyWorksJSON(`{
  "transDate": "2025-01-15",
  "enterDate": "20250115",
  "dueDate": "2025-02-15T10:30:00Z",
  "period": 202501,
  "transTime": "143000",
  "amount": 1500.50,
  "description": "Invoice payment"
}`);

// All dates are converted to YYYYMMDD
console.log(data.transDate);  // "20250115" (typed as YYYYMMDD)
console.log(data.enterDate);  // "20250115" (typed as YYYYMMDD)
console.log(data.dueDate);    // "20250215" (typed as YYYYMMDD)

// Period is typed correctly
console.log(data.period);     // 202501 (typed as Period)

// Time is converted to HHMMSS
console.log(data.transTime);  // "143000" (typed as HHMMSS)

// Other fields remain unchanged
console.log(data.amount);     // 1500.5
console.log(data.description); // "Invoice payment"
```

### Customizing JSON Parsing

Control which fields get converted:

```typescript
import { parseMoneyWorksJSON } from '@moneyworks/utilities';

// Selective conversion
const data1 = parseMoneyWorksJSON(jsonString, {
  convertDates: true,
  convertPeriods: false,  // Don't convert periods
  convertTimes: false     // Don't convert times
});

// Add custom date fields
const data2 = parseMoneyWorksJSON(jsonString, {
  additionalDateFields: ['customDate', 'expiryDate']
});

// Exclude specific fields
const data3 = parseMoneyWorksJSON(jsonString, {
  excludeFields: ['rawDate', 'legacyTime']
});

// Strict mode - throw on invalid values
try {
  const data4 = parseMoneyWorksJSON(jsonString, {
    strict: true
  });
} catch (e) {
  console.error("Invalid data in JSON");
}
```

## JSON Stringification

Format data for different purposes:

```typescript
import { 
  stringifyMoneyWorks,
  stringifyForAPI,
  stringifyForDisplay,
  createYYYYMMDD,
  createPeriod
} from '@moneyworks/utilities';

const data = {
  transDate: createYYYYMMDD("20250115"),
  period: createPeriod(202501),
  amount: 1500.50,
  description: "Payment"
};

// API format (compact, no formatting)
const apiJson = stringifyForAPI(data);
console.log(apiJson);
// {"transDate":"20250115","period":202501,"amount":1500.5,"description":"Payment"}

// Display format (formatted dates, pretty print)
const displayJson = stringifyForDisplay(data);
console.log(displayJson);
// {
//   "transDate": "2025-01-15",
//   "period": "2025/01",
//   "amount": 1500.5,
//   "description": "Payment"
// }

// Custom formatting
const customJson = stringifyMoneyWorks(data, {
  formatDates: true,
  dateSeparator: ".",
  formatPeriods: true,
  periodSeparator: "-",
  indent: 2
});
```

## Type Safety with Branded Types

Prevent common errors with branded types:

```typescript
import type { 
  YYYYMMDD, 
  AccountCode, 
  NameCode,
  TransactionSeq 
} from '@moneyworks/utilities';

// Define function with specific types
function createInvoice(
  date: YYYYMMDD,
  customer: NameCode,
  account: AccountCode,
  amount: number
) {
  // Implementation
}

// Type-safe usage
const invoiceDate = createYYYYMMDD("20250115");
const customer: NameCode = "CUST001" as NameCode;
const salesAccount: AccountCode = "4000" as AccountCode;

createInvoice(invoiceDate, customer, salesAccount, 1000);

// This won't compile - prevents parameter mixups
// createInvoice(customer, invoiceDate, salesAccount, 1000); // ❌ Type error

// Create your own branded types
import type { Brand } from '@moneyworks/utilities';

type InvoiceNumber = Brand<string, "InvoiceNumber">;
type OrderID = Brand<number, "OrderID">;

function processInvoice(invoice: InvoiceNumber, order: OrderID) {
  // Type-safe processing
}
```

## Date Arithmetic and Comparisons

Perform calculations with dates:

```typescript
import { 
  addDaysToYYYYMMDD,
  addMonthsToYYYYMMDD,
  daysBetweenYYYYMMDD,
  compareYYYYMMDD,
  todayYYYYMMDD,
  d
} from '@moneyworks/utilities';

const today = todayYYYYMMDD();
const invoiceDate = createYYYYMMDD("20250115");

// Add/subtract days
const dueDate = addDaysToYYYYMMDD(invoiceDate, 30);
const reminderDate = addDaysToYYYYMMDD(dueDate, -7);

// Add/subtract months
const nextQuarter = addMonthsToYYYYMMDD(invoiceDate, 3);
const lastYear = addMonthsToYYYYMMDD(invoiceDate, -12);

// Calculate days between
const daysUntilDue = daysBetweenYYYYMMDD(today, dueDate);
console.log(`Invoice due in ${daysUntilDue} days`);

// Compare dates
if (compareYYYYMMDD(today, dueDate) < 0) {
  console.log("Invoice not yet due");
}

// Or use the fluent API
const invoice = d`20250115`;
const due = invoice.addDays(30);

if (today.lt(due)) {
  const days = due.subtract(today);
  console.log(`${days} days until due`);
}
```

## Working with Business Days

Handle weekends and working days:

```typescript
import { d } from '@moneyworks/utilities';

function addWorkingDays(startDate: ReturnType<typeof d>, days: number) {
  let current = startDate;
  let workingDays = 0;
  
  while (workingDays < days) {
    current = current.addDays(1);
    if (!current.isWeekend()) {
      workingDays++;
    }
  }
  
  return current;
}

// Calculate delivery date
const orderDate = d`20250115`;
const deliveryDate = addWorkingDays(orderDate, 5);
console.log(`Delivery by: ${deliveryDate.format()}`);

// Check if date is weekend
const checkDate = d`20250118`; // Saturday
if (checkDate.isWeekend()) {
  console.log("Date falls on weekend");
}
```

## Month-End Operations

Work with month boundaries:

```typescript
import { 
  startOfMonthYYYYMMDD, 
  endOfMonthYYYYMMDD,
  d 
} from '@moneyworks/utilities';

const someDate = createYYYYMMDD("20250115");

// Get month boundaries
const monthStart = startOfMonthYYYYMMDD(someDate); // "20250101"
const monthEnd = endOfMonthYYYYMMDD(someDate);     // "20250131"

// Or with fluent API
const date = d`20250215`;
const isEndOfMonth = date.eq(endOfMonthYYYYMMDD(date));

// Handle month-end processing
function isLastBusinessDayOfMonth(date: YYYYMMDD): boolean {
  const end = endOfMonthYYYYMMDD(date);
  let lastBusiness = d`${end}`;
  
  while (lastBusiness.isWeekend()) {
    lastBusiness = lastBusiness.addDays(-1);
  }
  
  return date === lastBusiness.toString();
}
```

## Real-World Example: Invoice Aging

```typescript
import { d, todayYYYYMMDD } from '@moneyworks/utilities';

interface Invoice {
  number: string;
  date: YYYYMMDD;
  dueDate: YYYYMMDD;
  amount: number;
}

function categorizeInvoices(invoices: Invoice[]) {
  const today = d`${todayYYYYMMDD()}`;
  
  return invoices.map(invoice => {
    const invDate = d`${invoice.date}`;
    const dueDate = d`${invoice.dueDate}`;
    const age = today.subtract(invDate);
    
    let category: string;
    if (today.lte(dueDate)) {
      category = 'current';
    } else {
      const overdue = today.subtract(dueDate);
      if (overdue <= 30) category = '1-30 days';
      else if (overdue <= 60) category = '31-60 days';
      else if (overdue <= 90) category = '61-90 days';
      else category = 'over 90 days';
    }
    
    return {
      ...invoice,
      age,
      category,
      isOverdue: today.gt(dueDate),
      daysOverdue: Math.max(0, today.subtract(dueDate))
    };
  });
}
```
