## Complex Date Calculations

### Fiscal Year Handling

```typescript
import { d, p, PeriodUtils } from '@moneyworks/utilities';

interface FiscalConfig {
  yearStartMonth: number; // 1-12
  yearEndMonth: number;   // 1-12
}

class FiscalYearProcessor {
  constructor(private config: FiscalConfig) {}

  getFiscalYear(date: ReturnType<typeof d>): number {
    const period = date.toPeriod();
    const month = PeriodUtils.month(period);
    const year = PeriodUtils.year(period);
    
    // If date is before fiscal year start, it belongs to previous fiscal year
    return month < this.config.yearStartMonth ? year - 1 : year;
  }

  getFiscalQuarter(date: ReturnType<typeof d>): { quarter: number; year: number } {
    const period = date.toPeriod();
    const month = PeriodUtils.month(period);
    const fiscalYear = this.getFiscalYear(date);
    
    // Calculate months since fiscal year start
    const monthsFromStart = (month - this.config.yearStartMonth + 12) % 12;
    const quarter = Math.floor(monthsFromStart / 3) + 1;
    
    return { quarter, year: fiscalYear };
  }

  getFiscalPeriods(fiscalYear: number): Period[] {
    const startPeriod = PeriodUtils.fromYearMonth(
      fiscalYear, 
      this.config.yearStartMonth
    );
    const endPeriod = PeriodUtils.fromYearMonth(
      fiscalYear + 1, 
      this.config.yearStartMonth - 1
    );
    
    return PeriodUtils.range(startPeriod, endPeriod);
  }

  isYearEnd(date: ReturnType<typeof d>): boolean {
    const period = date.toPeriod();
    return PeriodUtils.month(period) === this.config.yearEndMonth;
  }
}

// Usage with April-March fiscal year
const fiscal = new FiscalYearProcessor({ 
  yearStartMonth: 4, 
  yearEndMonth: 3 
});

const transDate = d`20250215`;
const { quarter, year } = fiscal.getFiscalQuarter(transDate);
console.log(`FY${year} Q${quarter}`); // FY2024 Q4
```

### Payment Schedule Generator

```typescript
import { d, YYYYMMDDWithMethods } from '@moneyworks/utilities';

interface PaymentScheduleOptions {
  startDate: YYYYMMDDWithMethods;
  frequency: 'weekly' | 'fortnightly' | 'monthly' | 'quarterly' | 'annually';
  count: number;
  adjustForWeekends?: boolean;
  adjustForHolidays?: (date: YYYYMMDDWithMethods) => boolean;
}

class PaymentScheduleGenerator {
  generate(options: PaymentScheduleOptions): YYYYMMDDWithMethods[] {
    const payments: YYYYMMDDWithMethods[] = [];
    let current = options.startDate;
    
    for (let i = 0; i < options.count; i++) {
      // Calculate next payment date
      switch (options.frequency) {
        case 'weekly':
          current = i === 0 ? current : current.addDays(7);
          break;
        case 'fortnightly':
          current = i === 0 ? current : current.addDays(14);
          break;
        case 'monthly':
          current = i === 0 ? current : this.addMonthsSameDay(current, 1);
          break;
        case 'quarterly':
          current = i === 0 ? current : this.addMonthsSameDay(current, 3);
          break;
        case 'annually':
          current = i === 0 ? current : this.addMonthsSameDay(current, 12);
          break;
      }
      
      // Adjust for weekends and holidays
      if (options.adjustForWeekends || options.adjustForHolidays) {
        current = this.adjustDate(current, options);
      }
      
      payments.push(current);
    }
    
    return payments;
  }
  
  private addMonthsSameDay(date: YYYYMMDDWithMethods, months: number): YYYYMMDDWithMethods {
    const originalDay = parseInt(date.substring(6, 8));
    let result = date.addMonths(months);
    
    // Handle month-end scenarios (e.g., Jan 31 + 1 month)
    const resultDay = parseInt(result.substring(6, 8));
    if (resultDay !== originalDay) {
      // Go to last day of previous month
      result = d`${result.substring(0, 6)}01`.addDays(-1);
    }
    
    return result;
  }
  
  private adjustDate(
    date: YYYYMMDDWithMethods, 
    options: PaymentScheduleOptions
  ): YYYYMMDDWithMethods {
    let adjusted = date;
    
    // Move forward if weekend or holiday
    while (
      (options.adjustForWeekends && adjusted.isWeekend()) ||
      (options.adjustForHolidays && options.adjustForHolidays(adjusted))
    ) {
      adjusted = adjusted.addDays(1);
    }
    
    return adjusted;
  }
}

// Generate loan repayment schedule
const generator = new PaymentScheduleGenerator();
const schedule = generator.generate({
  startDate: d`20250131`,
  frequency: 'monthly',
  count: 12,
  adjustForWeekends: true,
  adjustForHolidays: (date) => {
    // Check if date is a public holiday
    const holidays = ['20250101', '20250425', '20251225'];
    return holidays.includes(date.toString());
  }
});

schedule.forEach((payment, i) => {
  console.log(`Payment ${i + 1}: ${payment.format()}`);
});
```

## Complex Type Safety Patterns

### Builder Pattern with Branded Types

```typescript
import type { 
  YYYYMMDD, 
  Period, 
  AccountCode, 
  NameCode, 
  Brand 
} from '@moneyworks/utilities';
import { d, p, createYYYYMMDD } from '@moneyworks/utilities';

// Define custom branded types
type InvoiceNumber = Brand<string, "InvoiceNumber">;
type LineItemID = Brand<number, "LineItemID">;

// Type-safe builder
class InvoiceBuilder {
  private invoice: Partial<Invoice> = {};
  
  setNumber(num: string): this {
    this.invoice.number = num as InvoiceNumber;
    return this;
  }
  
  setCustomer(code: string): this {
    this.invoice.customer = code as NameCode;
    return this;
  }
  
  setDates(invoice: YYYYMMDD | string, due?: YYYYMMDD | string): this {
    this.invoice.invoiceDate = typeof invoice === 'string' 
      ? createYYYYMMDD(invoice) 
      : invoice;
    
    this.invoice.dueDate = due 
      ? (typeof due === 'string' ? createYYYYMMDD(due) : due)
      : d`${this.invoice.invoiceDate}`.addDays(30);
    
    this.invoice.period = d`${this.invoice.invoiceDate}`.toPeriod();
    return this;
  }
  
  addLine(account: string, amount: number, gst: number): this {
    if (!this.invoice.lines) {
      this.invoice.lines = [];
    }
    
    this.invoice.lines.push({
      id: this.invoice.lines.length as LineItemID,
      account: account as AccountCode,
      amount,
      gst,
      total: amount + gst
    });
    
    return this;
  }
  
  build(): Invoice {
    if (!this.isValid()) {
      throw new Error('Invoice is incomplete');
    }
    
    const totalAmount = this.invoice.lines!.reduce((sum, line) => sum + line.amount, 0);
    const totalGst = this.invoice.lines!.reduce((sum, line) => sum + line.gst, 0);
    
    return {
      ...this.invoice,
      totalAmount,
      totalGst,
      totalDue: totalAmount + totalGst
    } as Invoice;
  }
  
  private isValid(): boolean {
    return !!(
      this.invoice.number && 
      this.invoice.customer && 
      this.invoice.invoiceDate &&
      this.invoice.lines && 
      this.invoice.lines.length > 0
    );
  }
}

interface Invoice {
  number: InvoiceNumber;
  customer: NameCode;
  invoiceDate: YYYYMMDD;
  dueDate: YYYYMMDD;
  period: Period;
  lines: Array<{
    id: LineItemID;
    account: AccountCode;
    amount: number;
    gst: number;
    total: number;
  }>;
  totalAmount: number;
  totalGst: number;
  totalDue: number;
}

// Usage
const invoice = new InvoiceBuilder()
  .setNumber('INV-2025-001')
  .setCustomer('CUST001')
  .setDates('20250115', '20250215')
  .addLine('4000', 1000, 100)
  .addLine('4100', 500, 50)
  .build();
```

### Type Guards and Validators

```typescript
import { 
  isYYYYMMDD, 
  isPeriod,
  YYYYMMDD,
  Period,
  AccountCode
} from '@moneyworks/utilities';

// Create type guard functions
function isAccountCode(value: unknown): value is AccountCode {
  return typeof value === 'string' && /^[0-9]{4}$/.test(value);
}

function isValidTransaction(data: unknown): data is Transaction {
  if (!data || typeof data !== 'object') return false;
  
  const obj = data as any;
  
  return (
    isYYYYMMDD(obj.transDate) &&
    isPeriod(obj.period) &&
    isAccountCode(obj.account) &&
    typeof obj.amount === 'number' &&
    typeof obj.gst === 'number'
  );
}

// Validation with detailed errors
class TransactionValidator {
  private errors: string[] = [];
  
  validate(data: unknown): data is Transaction {
    this.errors = [];
    
    if (!data || typeof data !== 'object') {
      this.errors.push('Data must be an object');
      return false;
    }
    
    const obj = data as any;
    
    if (!isYYYYMMDD(obj.transDate)) {
      this.errors.push('Invalid transDate: must be YYYYMMDD format');
    }
    
    if (!isPeriod(obj.period)) {
      this.errors.push('Invalid period: must be YYYYMM as number');
    }
    
    if (!isAccountCode(obj.account)) {
      this.errors.push('Invalid account: must be 4-digit code');
    }
    
    if (typeof obj.amount !== 'number' || obj.amount < 0) {
      this.errors.push('Invalid amount: must be positive number');
    }
    
    if (typeof obj.gst !== 'number' || obj.gst < 0) {
      this.errors.push('Invalid gst: must be positive number');
    }
    
    return this.errors.length === 0;
  }
  
  getErrors(): string[] {
    return this.errors;
  }
}

// Usage
const validator = new TransactionValidator();
const input = {
  transDate: '20250115',
  period: 202501,
  account: '4000',
  amount: 1000,
  gst: 100
};

if (validator.validate(input)) {
  // input is now typed as Transaction
  processTransaction(input);
} else {
  console.error('Validation errors:', validator.getErrors());
}
```

## Performance Optimization Patterns

### Batch Date Processing

```typescript
import { d, YYYYMMDDWithMethods } from '@moneyworks/utilities';

class DateBatchProcessor {
  private cache = new Map<string, YYYYMMDDWithMethods>();
  
  // Process dates in batches to leverage proxy caching
  processBatch(dates: string[]): Map<string, ProcessedDate> {
    const results = new Map<string, ProcessedDate>();
    
    // Pre-create all date objects to populate cache
    const dateObjects = dates.map(dateStr => {
      if (this.cache.has(dateStr)) {
        return this.cache.get(dateStr)!;
      }
      const dateObj = d`${dateStr}`;
      this.cache.set(dateStr, dateObj);
      return dateObj;
    });
    
    // Process in parallel where possible
    dateObjects.forEach((date, index) => {
      results.set(dates[index], {
        original: dates[index],
        formatted: date.format(),
        period: date.toPeriod(),
        dayOfWeek: date.toDate().getDay(),
        isWeekend: date.isWeekend(),
        quarter: Math.ceil(date.toPeriod() % 100 / 3),
        fiscalYear: this.getFiscalYear(date)
      });
    });
    
    return results;
  }
  
  private getFiscalYear(date: YYYYMMDDWithMethods): number {
    const period = date.toPeriod();
    const month = period % 100;
    const year = Math.floor(period / 100);
    return month >= 7 ? year + 1 : year;
  }
}

interface ProcessedDate {
  original: string;
  formatted: string;
  period: number;
  dayOfWeek: number;
  isWeekend: boolean;
  quarter: number;
  fiscalYear: number;
}

// Usage
const processor = new DateBatchProcessor();
const dates = Array.from({ length: 1000 }, (_, i) => 
  d`20250101`.addDays(i).toString()
);

const results = processor.processBatch(dates);
```

### Lazy Period Calculations

```typescript
import { Period, PeriodUtils } from '@moneyworks/utilities';

class LazyPeriodRange {
  private cache: Period[] | null = null;
  
  constructor(
    private start: Period,
    private end: Period
  ) {}
  
  *[Symbol.iterator](): Iterator<Period> {
    let current = this.start;
    while (current <= this.end) {
      yield current;
      current = PeriodUtils.addMonths(current, 1);
    }
  }
  
  toArray(): Period[] {
    if (!this.cache) {
      this.cache = [...this];
    }
    return this.cache;
  }
  
  map<T>(fn: (period: Period, index: number) => T): T[] {
    const results: T[] = [];
    let index = 0;
    for (const period of this) {
      results.push(fn(period, index++));
    }
    return results;
  }
  
  filter(predicate: (period: Period) => boolean): Period[] {
    const results: Period[] = [];
    for (const period of this) {
      if (predicate(period)) {
        results.push(period);
      }
    }
    return results;
  }
  
  find(predicate: (period: Period) => boolean): Period | undefined {
    for (const period of this) {
      if (predicate(period)) {
        return period;
      }
    }
    return undefined;
  }
}

// Usage - only computes what's needed
const range = new LazyPeriodRange(202401, 202512);

// Find first Q2 period
const firstQ2 = range.find(p => PeriodUtils.month(p) === 4);

// Get only year-end periods
const yearEnds = range.filter(p => PeriodUtils.month(p) === 12);

// Transform periods to date ranges
const dateRanges = range.map(period => ({
  period,
  start: PeriodUtils.startDate(period),
  end: PeriodUtils.endDate(period)
}));
```

class MoneyWorksAPIClient {

      method: 'POST',
      headers: { 'Content-Type': 'application/json' },

      asAt: YYYYMMDD;
      items: AgedReceivable[];

### Date Range Analytics

```typescript
import { d, YYYYMMDD, YYYYMMDDWithMethods } from '@moneyworks/utilities';

class DateRangeAnalytics {
  static analyzeDateDistribution(dates: YYYYMMDD[]): {
    byDayOfWeek: Map<number, number>;
    byMonth: Map<number, number>;
    byQuarter: Map<number, number>;
    weekendCount: number;
    averageDaysBetween: number;
  } {
    const byDayOfWeek = new Map<number, number>();
    const byMonth = new Map<number, number>();
    const byQuarter = new Map<number, number>();
    let weekendCount = 0;
    
    // Process each date
    const sortedDates = [...dates].sort();
    const dateObjects = sortedDates.map(date => d`${date}`);
    
    dateObjects.forEach(date => {
      // Day of week (0 = Sunday, 6 = Saturday)
      const dow = date.toDate().getDay();
      byDayOfWeek.set(dow, (byDayOfWeek.get(dow) || 0) + 1);
      
      // Month
      const month = date.toPeriod() % 100;
      byMonth.set(month, (byMonth.get(month) || 0) + 1);
      
      // Quarter
      const quarter = Math.ceil(month / 3);
      byQuarter.set(quarter, (byQuarter.get(quarter) || 0) + 1);
      
      // Weekend count
      if (date.isWeekend()) {
        weekendCount++;
      }
    });
    
    // Calculate average days between
    let totalDaysBetween = 0;
    for (let i = 1; i < dateObjects.length; i++) {
      totalDaysBetween += dateObjects[i].subtract(dateObjects[i - 1]);
    }
    const averageDaysBetween = dateObjects.length > 1 
      ? totalDaysBetween / (dateObjects.length - 1) 
      : 0;
    
    return {
      byDayOfWeek,
      byMonth,
      byQuarter,
      weekendCount,
      averageDaysBetween
    };
  }
  
  static findPatterns(dates: YYYYMMDD[]): {
    isWeekly: boolean;
    isMonthly: boolean;
    isSameDay: boolean;
    pattern?: string;
  } {
    if (dates.length < 2) {
      return { isWeekly: false, isMonthly: false, isSameDay: false };
    }
    
    const sortedDates = [...dates].sort();
    const intervals: number[] = [];
    
    // Calculate intervals
    for (let i = 1; i < sortedDates.length; i++) {
      intervals.push(
        d`${sortedDates[i]}`.subtract(d`${sortedDates[i - 1]}`)
      );
    }
    
    // Check for weekly pattern (7 days)
    const isWeekly = intervals.every(interval => interval === 7);
    
    // Check for monthly pattern (28-31 days)
    const isMonthly = intervals.every(interval => 
      interval >= 28 && interval <= 31
    );
    
    // Check if same day of month
    const daysOfMonth = sortedDates.map(date => 
      parseInt(date.substring(6, 8))
    );
    const isSameDay = daysOfMonth.every(day => day === daysOfMonth[0]);
    
    let pattern: string | undefined;
    if (isWeekly) pattern = 'weekly';
    else if (isMonthly && isSameDay) pattern = 'monthly (same day)';
    else if (isMonthly) pattern = 'monthly';
    
    return { isWeekly, isMonthly, isSameDay, pattern };
  }
}

// Usage example
const invoiceDates: YYYYMMDD[] = [
  '20250115', '20250215', '20250315', '20250415',
  '20250515', '20250615', '20250715', '20250815'
];

const analysis = DateRangeAnalytics.analyzeDateDistribution(invoiceDates);
console.log('Date Distribution Analysis:', analysis);

const patterns = DateRangeAnalytics.findPatterns(invoiceDates);
console.log('Pattern Detection:', patterns);
```
